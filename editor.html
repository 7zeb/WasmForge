async function loadFFmpeg() { try { console.log('Loading FFmpeg.wasm...'); const response = await fetch('https://cdn.jsdelivr.net/gh/ffmpegwasm/ffmpeg.wasm/dist/ffmpeg.wasm'); if (!response.ok) { throw new Error(`HTTP error! status: ${response.status}`); } const ffmpegWasm = await response.arrayBuffer(); console.log('FFmpeg.wasm loaded successfully!'); return ffmpegWasm; } catch (error) { console.error('Error loading FFmpeg.wasm:', error); throw new Error('Failed to load FFmpeg.wasm'); } }
